def parse_3ac(statement):
    # Remove extra spaces
    statement = statement.replace(" ", "")


    # Split into result and expression
    parts = statement.split('=')
    result = parts[0]
    expr = parts[1]


    # Operators to check
    operators = ['+', '-', '*', '/', '%']


    op = ''
    arg1 = ''
    arg2 = ''


    # Find operator inside expression
    for operator in operators:
        if operator in expr:
            op = operator
            arg1, arg2 = expr.split(operator)
            break


    # If no operator found (simple assignment)
    if op == '':
        op = '='
        arg1 = expr
        arg2 = ''


    return {
        'op': op,
        'arg1': arg1,
        'arg2': arg2,
        'result': result
    }




def main():
    n = int(input("Enter number of 3AC statements: "))
    statements = []


    for i in range(n):
        stmt = input(f"Enter 3AC statement #{i+1}: ")
        statements.append(parse_3ac(stmt))


    # Quadruples
    print("\n--- Quadruples ---")
    print(f"{'Op':<7}{'Arg1':<10}{'Arg2':<10}{'Result':<10}")


    for q in statements:
        print(f"{q['op']:<7}{q['arg1']:<10}{q['arg2']:<10}{q['result']:<10}")


    # Triples
    print("\n--- Triples ---")
    print(f"{'Index':<7}{'Op':<7}{'Arg1':<10}{'Arg2':<10}")


    for i, t in enumerate(statements):
        print(f"{i:<7}{t['op']:<7}{t['arg1']:<10}{t['arg2']:<10}")




if __name__ == "__main__":
    main()

def postfix(exp):
   s = []
   out = []


   for ch in exp:
       if ch.isalnum():
           out.append(ch)


       elif ch == '(':
           s.append(ch)


       elif ch == ')':
           while s[-1] != '(':
               out.append(s.pop())
           s.pop()


       else:
           while s and s[-1] != '(' and p[ch] <= p.get(s[-1], 0):
               out.append(s.pop())
           s.append(ch)
   while s:
       out.append(s.pop())


   return out

import re
def generate_code(statements):
   regs = {'R0': None, 'R1': None}
   addrs = {}
   ops = {'+': 'ADD', '-': 'SUB', '*': 'MUL', '/': 'DIV'}
  
   print("\n" + "=" * 50 + "\nCODE GENERATION TRACE\n" + "=" * 50)
  
   for i, stmt in enumerate(statements):
       #  1: 1-line Regex Unpacking (Handles spaces automatically!)
       x, y, op, z = re.match(r'(\w+)\s*=\s*(\w+)\s*([\+\-\*/])\s*(\w+)', stmt).groups()
       code = []
      
       #  2: 1-line Register Allocation using next()
       rx = addrs.get(y)
       if not rx or not rx.startswith('R'):
           rx = next((r for r, v in regs.items() if v is None), 'R0')
           code.append(f"MOV {y}, {rx}")
      
       # Generate Instruction
       code.append(f"{ops[op]} {addrs.get(z, z)}, {rx}")
      
       # Update Descriptors
       if regs[rx]: addrs.pop(regs[rx], None)
       regs[rx], addrs[x] = x, rx
      
       #  1-line Dictionary Filtering for Final Write-back Cleanup
       if i == len(statements) - 1:
           code.append(f"MOV {rx}, {x}")
           addrs[x] = f"{rx} and memory"
           regs = {r: (v if r == rx else None) for r, v in regs.items()}
           addrs = {k: v for k, v in addrs.items() if v.startswith(rx)}
          
       # Print output formatting natively
       print(f"Statement: {stmt}\nCode Generated:\n" + "\n".join(code))
       print("Register Descriptor:\n" + "\n".join(f"{r} contains {v}" for r, v in regs.items() if v))
       print("Address Descriptor:\n" + "\n".join(f"{k} in {v}" for k, v in addrs.items() if v))
       print("-" * 50)

def table():
   t = ['+', '-', '*', '/', 'id', '$']
   rel = {
       ('+', '+'): '>', ('+', '-'): '>', ('+', '*'): '<', ('+', '/'): '<', ('+', 'id'): '<', ('+', '$'): '>',
       ('-', '+'): '>', ('-', '-'): '>', ('-', '*'): '<', ('-', '/'): '<', ('-', 'id'): '<', ('-', '$'): '>',
       ('*', '+'): '>', ('*', '-'): '>', ('*', '*'): '>', ('*', '/'): '>', ('*', 'id'): '<', ('*', '$'): '>',
       ('/', '+'): '>', ('/', '-'): '>', ('/', '*'): '>', ('/', '/'): '>', ('/', 'id'): '<', ('/', '$'): '>',
       ('id', '+'): '>', ('id', '-'): '>', ('id', '*'): '>', ('id', '/'): '>', ('id', '$'): '>',
       ('$', '+'): '<', ('$', '-'): '<', ('$', '*'): '<', ('$', '/'): '<', ('$', 'id'): '<', ('$', '$'): 'acc'
   }
   return t, rel




def parse(exp):
   t, rel = table()
   stack = ['$']
   exp = exp + '$'


   tokens = []
   i = 0
   while i < len(exp):
       if exp[i:i+2] == 'id':
           tokens.append('id')
           i += 2
       else:
           tokens.append(exp[i])
           i += 1


   print("\nStack\t\tInput\t\tAction")
   print("-" * 40)


   while tokens:
       top = next((x for x in reversed(stack) if x in t), '$')
       cur = tokens[0]


       print("".join(stack), "\t\t", "".join(tokens), "\t", end="")


       if top == '$' and cur == '$':
           print("Accept")
           break


       action = rel.get((top, cur), '-')


       if action == '<':
           stack.append(tokens.pop(0))
           print("Shift")


       elif action == '>':
           if stack[-1] == 'id':
               stack.pop()
               stack.append('E')
               print("Reduce: id->E")
           elif len(stack) >= 3:
               op = stack[-2]
               stack[-3:] = ['E']
               print(f"Reduce: E {op} E -> E")


       else:
           print("Error")
           break

import re
import os


def optimize_code(lines):
    result = []
    aliases = {}       # Stores variable aliases (copy propagation)
    expr_map = {}      # Stores expressions for reuse (CSE)


    for line in lines:
        line = line.strip()


        # Skip empty lines or comments
        if not line or line.startswith("//") or '=' not in line:
            continue


        lhs, rhs = map(str.strip, line.split('=', 1))


        # --- 1. Copy Propagation ---
        for var, val in aliases.items():
            rhs = re.sub(rf'\b{var}\b', val, rhs)


        # If RHS is just a variable → store alias and skip output
        if re.fullmatch(r'[a-zA-Z_]\w*', rhs):
            aliases[lhs] = rhs
            continue


        # --- 2. Constant Folding ---
        if re.fullmatch(r'[\d\s+\-*/()]+', rhs):
            try:
                rhs = str(eval(rhs))
            except:
                pass  # Ignore invalid expressions safely


        # --- 3. Common Subexpression Elimination ---
        for expr in sorted(expr_map, key=len, reverse=True):
            if expr in rhs:
                rhs = rhs.replace(expr, expr_map[expr])


        expr_map[rhs] = lhs
        result.append(f"{lhs}={rhs}")


    return result

import re


# -------- Lexical Analysis --------
def lexer(expr):
   tokens = []
   sym = {}
   count = 1


   for t in re.findall(r'[a-zA-Z_]\w*|\d+\.\d+|\d+|[=+\-*()]', expr):
       if t.isalpha():
           if t not in sym:
               sym[t] = count
               count += 1
           tokens.append(("ID", t))
       elif t.isdigit():
           tokens.append(("NUM", t))
       else:
           tokens.append(("OP", t))


   return tokens, sym

class Node:
    def __init__(self, v, l=None, r=None):
        self.value, self.left, self.right = v, l, r


# Added only this (no printing)
def get_tokens(exp):
    return re.findall(r'\d+|[a-zA-Z_]\w*|[=+\-*/()]', exp)


def build_tree(tokens, p=0):
    node = Node(tokens.pop(0))
    prec = {'=':1,'+':2,'-':2,'*':3,'/':3}
    while tokens and prec.get(tokens[0],0) >= p:
        op = tokens.pop(0)
        node = Node(op, node, build_tree(tokens, prec[op] + (op!='=')))
    return node

def assemble_interactive():


    OPTAB = {
        "STOP": ("IS", "00"), "ADD": ("IS", "01"), "SUB": ("IS", "02"),
        "MULT": ("IS", "03"), "MOVER": ("IS", "04"), "MOVEM": ("IS", "05"),
        "READ": ("IS", "09"), "PRINT": ("IS", "10"),
        "START": ("AD", "01"), "END": ("AD", "02"),
        "DS": ("DL", "02"), "DC": ("DL", "01")
    }


    REG = {"AREG": "1", "BREG": "2", "CREG": "3", "DREG": "4"}


    print("Enter Assembly Code (END to stop):")
    lines = []
    while True:
        line = input().upper().strip()
        if not line: continue
        lines.append(line.replace(',', ' '))
        if "END" in line:
            break


    symtab = {}
    ic = []
    lc = 0


    # -------- PASS 1 --------
    for line in lines:
        tokens = line.split()


        # Label handling
        if tokens[0] not in OPTAB:
            symtab[tokens[0]] = lc
            tokens = tokens[1:]


        m = tokens[0]


        if m == "START":
            lc = int(tokens[1])
            ic.append(f"(AD,01) (C,{lc})")


        elif m == "END":
            ic.append("(AD,02)")


        elif m == "DS":
            ic.append(f"{lc} (DL,02) (C,{tokens[1]})")
            lc += int(tokens[1])


        elif m in OPTAB:
            entry = f"{lc} ({OPTAB[m][0]},{OPTAB[m][1]})"


            for t in tokens[1:]:
                if t in REG:
                    entry += f" ({REG[t]})"
                else:
                    entry += f" (S,{t})"
                    if t not in symtab:
                        symtab[t] = None


            ic.append(entry)
            lc += 1


    # -------- OUTPUT --------
    print("\nSYMBOL TABLE")
    for k, v in symtab.items():
        print(k, "=", v)


    print("\nINTERMEDIATE CODE")
    for line in ic:
        print(line)

MDT1, MDT2, MNT, ALA2 = [], [], [], []
ALA1 = set()


# Single reusable display function
def display(title, header, data):
    print(f"\n{title}\n{header}")
    for i, d in enumerate(data, 1):
        if isinstance(d, tuple): 
            print(i, d[0], d[1])
        else: 
            print(i, d)


# ================= PASS 1 =================
def pass1(prog):
    i = 0
    while i < len(prog):
        if prog[i].upper() == "MACRO":
            i += 1
            cleaned_header = prog[i].replace(",", " ")
            parts = cleaned_header.split()
            
            # Store MNT as a tuple: (macro_name, mdt_index)
            MNT.append((parts[0], len(MDT1) + 1))
            ALA1.update(parts[1:])
            
            MDT1.append(cleaned_header)
            i += 1
            while prog[i].upper() != "MEND":
                MDT1.append(prog[i])
                i += 1
            MDT1.append("MEND")
        i += 1
        
    print("\n----- PASS 1 -----")
    display("MNT", "Index\tName\tMDT_Index", MNT)
    display("MDT", "Index\tMacro Definition", MDT1)
    display("ALA (PASS 1)", "Index\tArgument", list(ALA1))


# ================= PASS 2 =================
def pass2(prog):
    skip = False
    for line in prog:
        up_line = line.upper()
        if up_line == "MACRO": skip = True; continue
        if up_line == "MEND": skip = False; continue
        if skip: continue


        parts = line.replace(",", " ").split()
        
        # Find matching macro in MNT
        mnt_entry = next((m for m in MNT if m[0].lower() == parts[0].lower()), None)
        
        if mnt_entry:
            ptr = mnt_entry[1] - 1
            header = MDT1[ptr].split()
            
            # Map parameters dynamically
            mapping = dict(zip(header[1:], parts[1:]))
            ALA2.extend(parts[1:len(header)])
            
            MDT2.append(line)
            ptr += 1
            while MDT1[ptr].upper() != "MEND":
                stmt = MDT1[ptr]
                for k, v in mapping.items():
                    stmt = stmt.replace(k, v)
                MDT2.append(stmt)
                ptr += 1
            MDT2.append("MEND")


    print("\n----- PASS 2 -----")
    display("MNT", "Index\tName\tMDT_Index", MNT)
    display("MDT", "Index\tMacro Definition", MDT2)
    display("ALA (PASS 2)", "Index\tArgument", ALA2)

if __name__ == "__main__":
